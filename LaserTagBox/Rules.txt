def flee
    exploreExplosiveBarrels
        if near Barrel 
            moveAway
            layDown
        
    explore ditches
        if ditch 
            moveTowards ditch, but not near barrel
            LayDown
    
        

def decideMove
    exploreFlags
        if OwnFlag == carried
            if distanceToEnemyFlag <= 10
                moveTowardsEnemyFlag
            else
                moveToOwnFlag
        else
            if Action Points == 0
                move towards Flag
            if Action Points >= 2
                if Stance != standing
                    standUp
                    move towards Flag
            if Action Points == 4
                if Stance != standing
                    standUp
                    move towards Flag
                    layDown
        if CarryFlag
            if enemyThere
                moveHome
                layDown
            else 
                moveHome
def maybeKill
    exploreEnemies
    if found
        if found >= 2
            flee
            Set Fleed = True 
        if actionPoints >= 8
            exploreExplosive
            ExplosiveNerarEnemie?
            shoot (Explosive or Enemie)
        else shoot

def Reload
    if remainingShots == 0
        reload
        
def shoot 
    if action Points >= 7
        layDown
    Tag
